112  // true if 2 segments intersect; intersection point in "out"
113  bool intersect(const segment_t& rhs, vert_t& out, vert_t::value_type& u) const
114  {
115  vert_t r = second - first;
116  vert_t s = rhs.second - rhs.first;
117  vert_t::value_type det = r.cross2d(s);
118 
119  if (equivalent(det, zero))
120  return false;
121 
122  vert_t diff = rhs.first - first;
123  u = diff.cross2d(s) / det;
124  vert_t::value_type v = diff.cross2d(r) / det;
125 
126  out = first + (r * u);
127  return (u > 0.0 && u < 1.0 && v > 0.0 && v < 1.0);
128  }

179  return i0;
180  if (equivalent(p.x(), p1.x(), e) &&
181  equivalent(p.y(), p1.y(), e))
182  return i1;
183  if (equivalent(p.x(), p2.x(), e) &&
184  equivalent(p.y(), p2.y(), e))

190  inline bool get_barycentric(const vert_t& p, vert_t& out) const
191  {
192  vert_t v0 = p1 - p0, v1 = p2 - p0, v2 = p - p0;
193  vert_t::value_type d00 = v0.dot2d(v0);
194  vert_t::value_type d01 = v0.dot2d(v1);
195  vert_t::value_type d11 = v1.dot2d(v1);
196  vert_t::value_type d20 = v2.dot2d(v0);
197  vert_t::value_type d21 = v2.dot2d(v1);
198  vert_t::value_type denom = d00 * d11 - d01 * d01;
199 
200  // means that one of more of the triangles points are coincident:
201  if (equivalent(denom, 0.0))
202  return false;
203 
204  out.y() = (d11*d20 - d01 * d21) / denom;
205  out.z() = (d00*d21 - d01 * d20) / denom;
206  out.x() = 1.0 - out.y() - out.z();
207 
208  return true;
209  }

285  // add new triangle to the mesh from 3 indices
286  UID add_triangle(int i0, int i1, int i2)
287  {
288  if (i0 == i1 || i1 == i2 || i2 == i0)
289  return -1;
290 
291  UID uid(uidgen++);
292  triangle_t tri;
293  tri.uid = uid;
294  tri.i0 = i0;
295  tri.i1 = i1;
296  tri.i2 = i2;
297  tri.p0 = get_vertex(i0);
298  tri.p1 = get_vertex(i1);
299  tri.p2 = get_vertex(i2);
300  tri.a_min[0] = std::min(tri.p0.x(), std::min(tri.p1.x(), tri.p2.x()));
301  tri.a_min[1] = std::min(tri.p0.y(), std::min(tri.p1.y(), tri.p2.y()));
302  tri.a_max[0] = std::max(tri.p0.x(), std::max(tri.p1.x(), tri.p2.x()));
303  tri.a_max[1] = std::max(tri.p0.y(), std::max(tri.p1.y(), tri.p2.y()));
304 
305  // "2d_degenerate" means that either a) at least 2 points are coincident, or
306  // b) at least two edges are basically coincident (in the XY plane)
307  constexpr vert_t::value_type E = 0.0005;
308  tri.is_2d_degenerate =
309  same_vert(tri.p0, tri.p1, E) ||
310  same_vert(tri.p1, tri.p2, E) ||
311  same_vert(tri.p2, tri.p0, E) ||
312  same_vert((tri.p1 - tri.p0).normalize2d(), (tri.p2 - tri.p0).normalize2d(), E) ||
313  same_vert((tri.p2 - tri.p1).normalize2d(), (tri.p0 - tri.p1).normalize2d(), E) ||
314  same_vert((tri.p0 - tri.p2).normalize2d(), (tri.p1 - tri.p2).normalize2d(), E);
315 
316  _triangles.emplace(uid, tri);
317  _spatial_index.Insert(tri.a_min, tri.a_max, uid);
318 
319  ++_num_edits;
320 
321  return uid;

408  // insert a segment into the mesh, cutting triangles as necessary
409  void insert(const segment_t& seg, int marker)
410  {
411  // search for possible intersecting triangles:
412  vert_t::value_type a_min[2];
413  vert_t::value_type a_max[2];
414  a_min[0] = std::min(seg.first.x(), seg.second.x());
415  a_min[1] = std::min(seg.first.y(), seg.second.y());
416  a_max[0] = std::max(seg.first.x(), seg.second.x());
417  a_max[1] = std::max(seg.first.y(), seg.second.y());
418  std::vector<UID> uids;
419 
420  _spatial_index.Search(
421  a_min, a_max,
422  [&uids](const UID& u) {
423  uids.push_back(u);
424  return true;
425  });
426 
427  // The working set of triangles which we will add to if we have
428  // to split triangles. Any triangle only needs to be split once,
429  // even if it gets intersected multiple times. That is because each
430  // split generates new traigles, and discards the original, and further
431  // splits will just happen on the new triangles later. (That's why
432  // every split operation is followed by a "continue" to short-circuit
433  // to loop)
434  vert_t::value_type E = EPSILON; // E = 1e-3;
435  std::list<UID> uid_list;
436  std::copy(uids.begin(), uids.end(), std::back_inserter(uid_list));
437  for (auto uid : uid_list)
438  {
439  triangle_t& tri = _triangles[uid];
440 
441  // skip triangles that are "degenerate" in 2D. We will keep them
442  // because they may NOT be degenerate in 3D (e.g. steep slopes).
443  if (tri.is_2d_degenerate)
444  continue;
445 
446  // first see if one of the endpoints falls within a triangle.
447  // if so, split the triangle into 2 or 3 new ones, and mark
448  // all of its verts as CONSTRAINT so they will not be subject
449  // to morphing.
450  if (tri.contains_2d(seg.first))
451  {
452  if (inside_split(tri, seg.first, &uid_list, marker))
453  continue;
454  }
455 
456  if (tri.contains_2d(seg.second))
457  {
458  if (inside_split(tri, seg.second, &uid_list, marker))
459  continue;
460  }
461 
462  // Next try to intersect the segment with a triangle edge.
463  // In each case, make sure the intersection point isn't
464  // coincident with an existing vertex (in which case ignore it).
465  // If we do split an edge, mark all of its verts as CONSTRAINT
466  // so they will not be subject to morphing; furthermore, convery
467  // any BOUNDARY markers to the new vert so we can maintain boundaries
468  // for skirt generation.
469  vert_t out;
470  vert_t::value_type u;
471  UID new_uid;
472  int new_i;
473 
474  //... WE SHOULD do the whole "update the marker" thing here too, I think ...
475  //... But it's causing an infinite loop ... gw
476 
477 
478  // does the segment cross first triangle edge?
479  segment_t edge0(tri.p0, tri.p1);
480  if (seg.intersect(edge0, out, u) && !tri.is_vertex(out, E))
481  {
482  int new_marker = marker;
483  if ((_markers[tri.i0] & _boundary_marker) && (_markers[tri.i1] & _boundary_marker))
484  new_marker |= _boundary_marker;
485 
486  new_i = get_or_create_vertex(out, new_marker);
487  if (new_i < 0)
488  return;
489 
490  int new_tris = 0;
491 
492  new_uid = add_triangle(new_i, tri.i2, tri.i0);
493  if (new_uid >= 0) {
494  _markers[tri.i2] |= _constraint_marker;
495  _markers[tri.i0] |= _constraint_marker;
496  uid_list.push_back(new_uid);
497  ++new_tris;
498  }
499 
500  new_uid = add_triangle(new_i, tri.i1, tri.i2);
501  if (new_uid >= 0) {
502  _markers[tri.i1] |= _constraint_marker;
503  _markers[tri.i2] |= _constraint_marker;
504  uid_list.push_back(new_uid);
505  ++new_tris;
506  }
507 
508  if (new_tris > 0)
509  {
510  if (marker_not_set(new_i, _has_elevation_marker) &&
511  marker_is_set(tri.i0, _has_elevation_marker) && marker_is_set(tri.i1, _has_elevation_marker))
512  {
513  auto z0 = get_vertex(tri.i0).z(), z1 = get_vertex(tri.i1).z();
514  get_vertex(new_i).z() = z0 + u * (z1 - z0);
515  set_marker(new_i, _has_elevation_marker);
516  }
517 
518  remove_triangle(tri);
519  continue;
520  }
521  }
522 
523  // does the segment cross second triangle edge?
524  segment_t edge1(tri.p1, tri.p2);
525  if (seg.intersect(edge1, out, u) && !tri.is_vertex(out, E))
526  {
527  int new_marker = marker;
528  if ((_markers[tri.i1] & _boundary_marker) && (_markers[tri.i2] & _boundary_marker))
529  new_marker |= _boundary_marker;
530 
531  new_i = get_or_create_vertex(out, new_marker);
532  if (new_i < 0)
533  return;
534 
535  int new_tris = 0;
536 
537  new_uid = add_triangle(new_i, tri.i0, tri.i1);
538  if (new_uid >= 0) {
539  _markers[tri.i0] |= _constraint_marker;
540  _markers[tri.i1] |= _constraint_marker;
541  uid_list.push_back(new_uid);
542  ++new_tris;
543  }
544 
545  new_uid = add_triangle(new_i, tri.i2, tri.i0);
546  if (new_uid >= 0) {
547  _markers[tri.i2] |= _constraint_marker;
548  _markers[tri.i0] |= _constraint_marker;
549  uid_list.push_back(new_uid);
550  ++new_tris;
551  }
552 
553  if (new_tris > 0)
554  {
555  if (marker_not_set(new_i, _has_elevation_marker) &&
556  marker_is_set(tri.i1, _has_elevation_marker) && marker_is_set(tri.i2, _has_elevation_marker))
557  {
558  auto z0 = get_vertex(tri.i1).z(), z1 = get_vertex(tri.i2).z();
559  get_vertex(new_i).z() = z0 + u * (z1 - z0);
560  set_marker(new_i, _has_elevation_marker);
561  }
562 
563  remove_triangle(tri);
564  continue;
565  }
566  }
567 
568  // does the segment cross third triangle edge?
569  segment_t edge2(tri.p2, tri.p0);
570  if (seg.intersect(edge2, out, u) && !tri.is_vertex(out, E))
571  {
572  int new_marker = marker;
573  if ((_markers[tri.i2] & _boundary_marker) && (_markers[tri.i0] & _boundary_marker))
574  new_marker |= _boundary_marker;
575 
576  new_i = get_or_create_vertex(out, new_marker);
577  if (new_i < 0)
578  return;
579 
580  int new_tris = 0;
581 
582  new_uid = add_triangle(new_i, tri.i1, tri.i2);
583  if (new_uid >= 0) {
584  _markers[tri.i1] |= _constraint_marker;
585  _markers[tri.i2] |= _constraint_marker;
586  uid_list.push_back(new_uid);
587  ++new_tris;
588  }
589 
590  new_uid = add_triangle(new_i, tri.i0, tri.i1);
591  if (new_uid >= 0) {
592  _markers[tri.i0] |= _constraint_marker;
593  _markers[tri.i1] |= _constraint_marker;
594  uid_list.push_back(new_uid);
595  ++new_tris;
596  }
597 
598  if (new_tris > 0)
599  {
600  if (marker_not_set(new_i, _has_elevation_marker) &&
601  marker_is_set(tri.i2, _has_elevation_marker) && marker_is_set(tri.i0, _has_elevation_marker))
602  {
603  auto z0 = get_vertex(tri.i2).z(), z1 = get_vertex(tri.i0).z();
604  get_vertex(new_i).z() = z0 + u * (z1 - z0);
605  set_marker(new_i, _has_elevation_marker);
606  }
607 
608  remove_triangle(tri);
609  continue;
610  }
611  }
612  }

615  // inserts point "p" into the interior of triangle "tri",
616  // adds three new triangles, and removes the original triangle.
617  // return true if a split actual happened
618  bool inside_split(triangle_t& tri, const vert_t& p, std::list<UID>* uid_list, int new_marker)
619  {
620  int new_i = get_or_create_vertex(p, new_marker);
621  if (new_i < 0)
622  return false;
623 
624  // if this vertex is already one of the triangle's verts, no split is necessary
625  // and we are done.
626  if (tri.is_vertex(new_i))
627  return false;
628 
629  UID new_uid;
630  int new_tris = 0;
631 
632  // calculate the barycenric coordinates of the new point
633  // within the target triangle so we can detect any split
634  // that occurs right on an edge and then bypass creating
635  // a degenerate triangle.
636  vert_t bary(1, 1, 1);
637  if (tri.get_barycentric(p, bary) == false)
638  return false;
639 
640  if (!equivalent(bary[2], 0.0, EPSILON)) {
641  new_uid = add_triangle(tri.i0, tri.i1, new_i);
642  if (new_uid >= 0 && uid_list) {
643  _markers[tri.i0] |= _constraint_marker;
644  _markers[tri.i1] |= _constraint_marker;
645  uid_list->push_back(new_uid);
646  ++new_tris;
647  }
648  }
649 
650  if (!equivalent(bary[0], 0.0, EPSILON)) {
651  new_uid = add_triangle(tri.i1, tri.i2, new_i);
652  if (new_uid >= 0 && uid_list) {
653  _markers[tri.i1] |= _constraint_marker;
654  _markers[tri.i2] |= _constraint_marker;
655  uid_list->push_back(new_uid);
656  ++new_tris;
657  }
658  }
659 
660  if (!equivalent(bary[1], 0.0, EPSILON)) {
661  new_uid = add_triangle(tri.i2, tri.i0, new_i);
662  if (new_uid >= 0 && uid_list) {
663  _markers[tri.i2] |= _constraint_marker;
664  _markers[tri.i0] |= _constraint_marker;
665  uid_list->push_back(new_uid);
666  ++new_tris;
667  }
668  }
669 
670  if (new_tris > 0)
671  {
672  // if all verts in the surrounding triangle have elevation data,
673  // but this new one doesn't, set it via interpolation.
674  if (marker_not_set(new_i, _has_elevation_marker) &&
675  marker_is_set(tri.i0, _has_elevation_marker) &&
676  marker_is_set(tri.i1, _has_elevation_marker) &&
677  marker_is_set(tri.i2, _has_elevation_marker))
678  {
679  // interpolate the Z value from the triangle:
680  get_vertex(new_i).z() =
681  get_vertex(tri.i0).z() * bary[0] +
682  get_vertex(tri.i1).z() * bary[1] +
683  get_vertex(tri.i2).z() * bary[2];
684 
685  set_marker(new_i, _has_elevation_marker);
686  }
687 
688  remove_triangle(tri);
689  }
690 
691  return (new_tris > 0);
692  }

694  vert_t point_on_edge_closest_to(const edge_t& edge, const vert_t& p) const {
695  const vert_t& e1 = get_vertex(edge._i0);
696  const vert_t& e2 = get_vertex(edge._i1);
697  vert_t qp = e2 - e1;
698  vert_t xp = p - e1;
699  double u = xp.dot2d(qp) / qp.dot2d(qp);
700  if (u < 0.0) return e1;
701  else if (u > 1.0) return e2;
702  else return e1 + qp * u;
703  }

858  // attempt to find the concave hull by walking the outside of subgraph.
859  void get_hull(int graphid, const mesh_t& mesh, std::vector<int>& hull) const
860  {
861  const node_t* start = nullptr;
862  vert_t::value_type min_y = DBL_MAX;
863  vert_t::value_type min_x = DBL_MAX;
864 
865  // find the vertex with minimum y in graph graphid.
866  // TODO: it would be faster to find a vertex in graph graphid,
867  // and then traverse from there.
868  for (const auto& node_iter : _nodes)
869  {
870  const node_t& node = node_iter.second;
871  if (node._graphid == graphid)
872  {
873  const vert_t& vert = mesh.get_vertex(node._vertex_index);
874  if (start == nullptr || vert.y() < min_y)
875  {
876  start = &node;
877  min_y = vert.y();
878  }
879  }
880  }
881 
882  if (start == nullptr)
883  return;
884 
885  // next walk the boundary in a clockwise direction
886  const node_t* curr_node = start;
887  const vert_t& first = mesh.get_vertex(curr_node->_vertex_index);
888  const node_t* prev_node = nullptr;
889  vert_t prev_vert(first + vert_t(0, -1, 0));
890 
891  std::set<const node_t*> visited;
892 
893  while (true)
894  {
895  hull.push_back(curr_node->_vertex_index);
896 
897  double best_score = DBL_MAX;
898  const node_t* best_edge = nullptr;
899  const vert_t& curr_vert = mesh.get_vertex(curr_node->_vertex_index);
900 
901  vert_t invec = (curr_vert - prev_vert).normalize2d();
902 
903  for (auto& edge : curr_node->_edges)
904  {
905  if (visited.find(edge) != visited.end())
906  continue;
907  //if (edge == prev_node) // no backtracking
908  // continue;
909 
910  const vert_t& next_vert = mesh.get_vertex(edge->_vertex_index);
911  vert_t outvec = (next_vert - curr_vert).normalize2d();
912 
913  double turn = invec.cross2d(outvec) > 0.0 ? -1.0 : 1.0;
914  double dot = 1.0 - (0.5*(invec.dot2d(outvec) + 1.0));
915  double score = turn * dot;
916 
917  if (score < best_score)
918  {
919  best_score = score;
920  best_edge = edge;
921  }
922  }
923 
924  if (best_edge == nullptr)
925  {
926  ROCKY_WARN << "got stuck! :(" << std::endl;
927  break;
928  }
929 
930  if (best_edge == start)
931  {
932  // done!
933  break;
934  }
935 
936  if (curr_node != start)
937  visited.insert(curr_node);
938 
939  prev_node = curr_node;
940  prev_vert = curr_vert;
941  curr_node = best_edge;
942  }
943  }
944  };
