97  bool SurfaceNode::isVisible(vsg::State* state) const
98  {
99  // bounding box visibility check; this is much tighter than the bounding
100  // sphere. _frustumStack.top() contains the frustum in world coordinates.
101  // https://github.com/vsg-dev/VulkanSceneGraph/blob/master/include/vsg/vk/State.h#L267
102  // The first 8 points in _worldPoints are the 8 corners of the bounding box
103  // in world coordinates.
104  // Note: POLYTOPE_SIZE is defined in vsg plane.h
105  auto& frustum = state->_frustumStack.top();
106  int p;
107  for (int f = 0; f < POLYTOPE_SIZE; ++f) {
108  for (p = 0; p < 8; ++p)
109  if (vsg::distance(frustum.face[f], _worldPoints[p]) > 0.0) // visible?
110  break;
111  if (p == 8)
112  return false;
113  }
114 
115  // still good? check against the horizon.
116  shared_ptr<Horizon> horizon;
117  if (state->getValue("horizon", horizon))
118  {
119  for (p = 0; p < 4; ++p)
120  {
121  auto& wp = _worldPoints[p];
122  if (horizon->isVisible(wp.x, wp.y, wp.z))
123  return true;
124  }
125 
126  return false;
127  }
128  else
129  {
130  return true;
131  }
132  }
