105  inline vsg::ref_ptr<vsg::Data> moveImageData(shared_ptr<Image> image)
106  {
107  if (!image) return { };
108 
109  switch (image->pixelFormat())
110  {
111  case Image::R8_UNORM:
112  return move<unsigned char>(image, VK_FORMAT_R8_UNORM);
113  break;
114  case Image::R8G8_UNORM:
115  return move<vsg::ubvec2>(image, VK_FORMAT_R8G8_UNORM);
116  break;
117  case Image::R8G8B8_UNORM:
118  return move<vsg::ubvec3>(image, VK_FORMAT_R8G8B8_UNORM);
119  break;
120  case Image::R8G8B8A8_UNORM:
121  return move<vsg::ubvec4>(image, VK_FORMAT_R8G8B8A8_UNORM);
122  break;
123  case Image::R16_UNORM:
124  return move<unsigned short>(image, VK_FORMAT_R16_UNORM);
125  break;
126  case Image::R32_SFLOAT:
127  return move<float>(image, VK_FORMAT_R32_SFLOAT);
128  break;
129  case Image::R64_SFLOAT:
130  return move<double>(image, VK_FORMAT_R64_SFLOAT);
131  break;
132  };
133 
134  return { };
135  }

153  // Convert a vsg::Data structure to an Image if possible
154  inline Result<shared_ptr<Image>> makeImageFromVSG(vsg::ref_ptr<vsg::Data> data)
155  {
156  if (!data)
157  return Status(Status::ResourceUnavailable);
158 
159  // TODO: move this into a utility somewhere
160  auto vkformat = data->properties.format;
161 
162  Image::PixelFormat format =
163  vkformat == VK_FORMAT_R8_UNORM ? Image::R8_UNORM :
164  vkformat == VK_FORMAT_R8G8_UNORM ? Image::R8G8_UNORM :
165  vkformat == VK_FORMAT_R8G8B8_UNORM ? Image::R8G8B8_UNORM :
166  vkformat == VK_FORMAT_R8G8B8A8_UNORM ? Image::R8G8B8A8_UNORM :
167  vkformat == VK_FORMAT_R16_UNORM ? Image::R16_UNORM :
168  vkformat == VK_FORMAT_R32_SFLOAT ? Image::R32_SFLOAT :
169  vkformat == VK_FORMAT_R64_SFLOAT ? Image::R64_SFLOAT :
170  Image::UNDEFINED;
171 
172  if (format == Image::UNDEFINED)
173  {
174  return Status(Status::ResourceUnavailable, "Unsupported image format");
175  }
176 
177  auto image = Image::create(
178  format,
179  data->width(),
180  data->height(),
181  data->depth());
182 
183  memcpy(image->data<uint8_t>(), data->dataPointer(), image->sizeInBytes());
184 
185  if (data->properties.origin == vsg::TOP_LEFT)
186  {
187  image->flipVerticalInPlace();
188  }
189 
190  return Result(image);
191  }
