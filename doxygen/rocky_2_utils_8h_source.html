354  inline void* use(const T& data, void* token)
355  {
356  // Find the tracker for this tile and update its timestamp
357  if (token)
358  {
359  Token* ptr = static_cast<Token*>(token);
360 
361  // Move the tracker to the front of the list (ahead of the sentry).
362  // Once a cull traversal is complete, all visited tiles will be
363  // in front of the sentry, leaving all non-visited tiles behind it.
364  _list.splice(_list.begin(), _list, *ptr);
365  *ptr = _list.begin();
366  return ptr;
367  }
368  else
369  {

371  Token* ptr = new Token();
372  _list.emplace_front(data, ptr); // ListEntry

378  inline void flush(
379  unsigned maxCount,
380  std::function<bool(T& obj)> dispose)
381  {
382  // After cull, all visited tiles are in front of the sentry, and all
383  // non-visited tiles are behind it. Start at the sentry position and
384  // iterate over the non-visited tiles, checking them for deletion.
385  ListIterator i = _sentryptr;
386  ListIterator tmp;
387  unsigned count = 0;
388 
389  for (++i; i != _list.end() && count < maxCount; ++i)
390  {
391  ListEntry& le = *i;
392 
393  bool disposed = true;
394 
395  // user disposal function
396  if (dispose != nullptr)
397  disposed = dispose(le._data);
398 
399  if (disposed)
400  {
401  // back up the iterator so we can safely erase the entry:
402  tmp = i;
403  --i;
404 
405  // delete the token
406  delete static_cast<Token*>(le._token);
407 
408  // remove it from the tracker list:
409  _list.erase(tmp);
410  ++count;
411  }
412  }
413 
414  // reset the sentry.
415  _list.splice(_list.begin(), _list, _sentryptr);
416  _sentryptr = _list.begin();
417  }

493  template<class K, class V>
494  class LRUCache
495  {
496  private:
497  mutable std::mutex mutex;
498  int capacity;
499  using E = typename std::pair<K, V>;
500  typename std::list<E> cache;
501  std::unordered_map<K, typename std::list<E>::iterator> map;
502 
503  public:
504  int hits = 0;
505  int gets = 0;
506 
507  LRUCache(int capacity = 32) : capacity(capacity) { }
508 
509  inline void setCapacity(int value) {
510  std::scoped_lock L(mutex);
511  cache.clear();
512  map.clear();
513  hits = 0;
514  gets = 0;
515  capacity = std::max(0, value);
516  }
517 
518  inline V get(const K& key) {
519  if (capacity == 0) return V();
520  std::scoped_lock L(mutex);
521  ++gets;
522  auto it = map.find(key);
523  if (it == map.end())
524  return V();
525  cache.splice(cache.end(), cache, it->second);
526  ++hits;
527  return it->second->second;
528  }
529 
530  inline void put(const K& key, const V& value) {
531  if (capacity == 0) return;
532  std::scoped_lock L(mutex);
533  if (cache.size() == capacity) {
534  auto first_key = cache.front().first;
535  cache.pop_front();
536  map.erase(first_key);
537  }
538  cache.push_back({ key, value });
539  map[key] = --cache.end();
540  }
